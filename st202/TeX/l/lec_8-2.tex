\subsection{Covariance and Correlation}
\vskip.2cm

\begin{definition}
Let $X,Y$ be random variables. The \textbf{covariance} of $X$ and $Y$ is defined as
\begin{align*}
    \Cov(X,Y)&=\mathbb E[(X-\mathbb E(x))(Y-\mathbb E(Y))] \\
    &= \mathbb E(XY)-\mathbb E(X)\mathbb E(Y).
\end{align*}
\end{definition}
Some helpful properties of covariance:
\begin{itemize}
\item $\Cov(aX,aY)=ab\text{Cov}(X,Y)$
\item $\Cov(X+c , \ Y+d)=\text{Cov}(X,Y)$
\item $\Cov(X,X)=\Var(X)$
\item $\Cov(X+Y,\, U+V)=\Cov(X,U)+\text{Cov}(X,V)+\text{Cov}(Y,U)+\text{Cov}(Y,V)$
\end{itemize}

\begin{definition}
Let $X,Y$ be random variables. The \textbf{correlation coefficient} of $X$ and $Y$ is

$$\Corr(X,Y)=\frac{\Cov(X,Y)}{\sqrt{\Var(X)\Var(Y)}}=\rho,$$
with $-1\leq \rho \leq 1.$
\end{definition}
